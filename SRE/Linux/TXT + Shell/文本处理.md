> 文本内容的编辑与处理

------

### vim文本编辑

```bash
Insert:按 i/a/o 进入（写内容）
=========================================
Normal:按 Esc 进入（浏览、删除、复制、粘贴）
# 浏览
gg                    # 跳到文件第一行（看配置文件开头的全局变量）
G                     # 跳到文件最后一行（看日志末尾）
:123                  # 跳转到第123行（看报错行号时精准定位）
0                     # 跳到行首（光标所在行的第一个字符）
^                     # 跳到行首第一个非空字符（比0更常用）
$                     # 跳到行尾
Ctrl + d              # 向下翻半页
Ctrl + u              # 向上翻半页
# 搜索
/error                # 向下搜索 "error"，按 n 跳到下一个，按 N 跳到上一个
?error                # 向上搜索
# 操作
x                     # 删除字符
dd                    # 删除整行（剪切）
yy                    # 复制当前行
p                     # 粘贴
u                     # 撤销
Ctrl + r              # 撤销撤销
===========================================
Command在正常模式下按:进入（保存、退出、搜索替换）
# 退出
ZZ                    # 保存并退出
ZQ                    # 强制退出
:w (/tmp/backup.conf) # 保存(为备份文件)
# 设置
:set nu                # 显示行号（查报错行号，如 Nginx 配置报错）
:set nonu              # 取消行号
:set paste             # 进入粘贴模式（线上改 YAML 必须开！关闭自动缩进，防止格式错乱）
:set nopaste           # 退出粘贴模式（改完立马关）
:set fileencoding      # 查看当前文件编码（看是不是 UTF-8）
:set ff=unix           # 将换行符转为 Unix 格式（解决 Windows 传上来的 ^M 乱码）
:syntax on             # 开启语法高亮（看 JSON/YAML 结构是否错误）
:e ++enc=utf-8         # 用 UTF-8 编码重新加载文件（解决中文乱码）
# 替换
:%s/old/new/g          # 将全文的 old 替换为 new（不加 g 只换第一个）
:%s/old/new/gc         # 交互式替换（逐个确认，改核心配置强烈建议加 c）
:10,20s/old/new/g      # 仅替换第 10 行到第 20 行（精准打击）
:%s/^/# /g             # 全文行首加注释（#后面加空格，适合批量注释 Shell）
:%s/^# //g             # 批量去掉行首的 # 注释
:%s/\s\+$//g           # 删除所有行尾多余的空格（清理 YAML 和 Python 必备）
# 多窗口
:e /etc/hosts          # 在当前 Vim 中打开另一个文件（不退出当前编辑）
:bn                    # 切换到下一个文件（buffer next）
:bp                    # 切换到上一个文件
:ls                    # 列出当前所有打开的文件缓冲区
:sp filename           # 水平切分窗口（上下分屏，对比两个配置文件）
:vsp filename          # 垂直切分窗口（左右分屏）
Ctrl + ww              # 在分屏窗口间快速切换
:q                     # 关闭当前窗口
# 多列修改
# 操作流程（以给第10-20行加 # 注释为例）：
# 1. 按 Esc 确保正常模式
# 2. 输入 :10 回车（跳到第10行）
# 3. 按 Ctrl + v （进入列选择）
# 4. 按 10j   （向下移动10行，选中列）
# 5. 按 Shift + i （进入插入模式，光标在列首）
# 6. 输入 #    （写注释符）
# 7. 按 Esc   （等待1秒，所有选中的行首都会加上 #）
```

### 管道与重定向

```Bash
# 重定向（把命令的“输入/输出”从屏幕改成“文件”）
输入 >  a.txt        # 把结果写进文件（覆盖原有内容）
输入 >> a.txt        # 把结果追加到文件末尾（不覆盖）
命令 <  a.txt        # 把文件内容作为命令的输入
命令 2> a.txt        # 把报错信息写进文件（1是正常结果，2是报错）
命令 > a.txt 2>&1    # 把报错当成正常结果一起处理（常用！）

# 管道（把左边的输出，传给右边当输入）
命令1 | (命令1的输出->)命令2           # grep,awk,sed,wc,sort,uniq,head,tail
命令1 | xargs (命令1的输出->)命令2     # rm,cp,mv,mkdir,chmod,chown,kill,tar
命令1 | tee 文件                     # 屏幕显示结果，同时自动存进文件（查实时日志必备）
```

### 文本内容查看

```shell
# 查看或拼接文件内容
cat [选项] 文件...
cat -n            # 对所有输出行编号（含空行）
cat -b            # 对非空输出行编号
cat -s            # 将连续的空行压缩为一行
cat -T            # 将制表符显示为 ^I
cat -E            # 在每行结尾显示 $
tac [选项] 文件... # 反向输出文件内容

# 分页查看大文件内容
less [选项] 文件...
less -N           # 显示行号
less -S           # 不换行显示，可左右滚动
less -R           # 保留颜色转义序列（如日志高亮）
less -g           # 高亮显示上次搜索的关键词
# 交互快捷键：空格（下翻一页）、b（上翻一页）、/关键词（搜索）、q（退出）

# 显示文件开头部分（默认前 10 行）
head [选项] 文件...
head -n N          # 显示前 N 行（如 -n 20）
head -c N          # 显示前 N 个字节（如 -c 1024）
head -q            # 多个文件时不显示文件名头

# 显示文件末尾部分（默认后 10 行，运维最常用）
tail [选项] 文件...
tail -n N          # 显示后 N 行（如 -n 100）
tail -f            # 实时跟踪文件新增内容（日志监控神器）
tail -F            # 类似 -f，但文件被轮转后能重新打开（推荐用 -F）
tail -c N          # 显示后 N 个字节
tail --pid=PID     # 当指定进程结束时，自动终止 tail -f
```

### 搜索与过滤文本

```Shell
# 从文件的每一行中提取文本片段
cut [选项] 文件...
cut -d ':,'        # 指定字段分隔符
cut -f 2,4         # 提取第 N 列（如 1,3 或 2-5）
cut -c 1-10        # 提取每行 n-m 个字符（如 1-10）

# 在文件或标准输入中搜索匹配文本的行（文本过滤核心工具）
grep [选项] 模式 [文件...]
grep -i             # 忽略大小写
grep -v             # 反向匹配（显示不包含模式的行）
grep -r             # 递归搜索目录下的所有文件
grep -n             # 显示匹配行在文件中的行号
grep -c             # 只统计匹配的行数（不显示具体行）
grep -E             # 使用扩展正则表达式
grep -A/B/C N       # 显示匹配行及其后/前/前后 N 行
grep --color=auto   # 高亮显示匹配的字符串

# 流编辑器，对文件进行文本替换、删除、插入等操作（非交互）
sed [选项] '脚本' [文件...]
sed -i              # 直接修改源文件（慎用，建议先备份）
sed -i.bak          # 修改前自动备份为 .bak 后缀文件
sed -n              # 取消默认输出（常与 p 命令配合，只打印匹配行）
sed -e              # 允许执行多个脚本命令
's/old/new/g'       # 全局替换（g 代表全部替换）
'd'                 # 删除匹配行
'p'                 # 打印行（需配合 -n）
'N 2,5d'            # 范围：删除第2到第5行

# 强大的文本分析工具，按列处理（取日志字段、统计等）
awk '模式 {动作}' [文件...]
awk -F              # 指定输入字段分隔符（如 -F: 指定冒号分隔）
awk -v              # 定义变量并传入 awk 内部
'$0'                # 整行内容
'$1-$N'             # 第 1 到第 N 列
'NF'                # 当前行的字段总数（$NF 表示最后一列）
'NR'                # 当前行号
```

### 编辑与转换文本

```Bash
# 统计文件的行数、单词数、字符数
wc [选项] 文件...
wc -l             # 只统计行数（最常用）
wc -w             # 只统计单词数
wc -c             # 只统计字节数
wc -m             # 只统计字符数（与 -c 区别在于多字节字符）
wc -L             # 显示最长行的长度

# 对文本文件内容进行排序
sort [选项] 文件...
sort -n            # 按数值大小排序（而非字典序）
sort -r            # 反向排序（从大到小）
sort -t            # 指定字段分隔符（如 -t:）
sort -k N          # 按第 N 个字段排序（如 -k 3）
sort -u            # 去重（相同行只保留一行）
sort -h            # 对人类可读数字排序（如 1K, 2M）

# 报告或忽略重复行（通常与 sort 配合使用，处理相邻重复）
uniq [选项] [输入文件] [输出文件]
uniq -c             # 在每行前显示该行重复出现的次数
uniq -d             # 只显示重复行（且只显示一次）
uniq -u             # 只显示不重复的行（出现一次的行）
uniq -i             # 忽略大小写比较

# 替换、删除或压缩标准输入中的字符
tr [选项] 字符集1 [字符集2]
tr -d                # 删除所有匹配字符集1的字符
tr -s                # 压缩重复出现的连续字符为单个
tr -c                # 补集，匹配字符集1以外的所有字符
tr -t                # 截断字符集1长度与字符集2相同
```

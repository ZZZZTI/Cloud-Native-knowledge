### 用户与组管理

```bash
# 查看用户信息
id [用户名]              # 查看用户UID、GID及所属组(系统用户:UID=1~999,普通用户:UID>=1000)
whoami                  # 显示当前用户名
who/w                   # 详细登录用户信息
last/lastlog            # 查看历史登录记录/用户最后登录时间
su [-] 用户名            # 切换用户（- 加载目标用户环境变量）
groups username         # 查看指定用户所属组
getent passwd 用户名     # 从数据库获取用户信息
getent group 组名        # 获取组信息

# 修改用户信息
useradd [选项] 用户名     # 创建用户
-u UID, -g/G 组名, -m(生成家目录), -s(shell), -r(系统用户), -c "xxx"(用户描述)
usermod [选项] 用户名     # 修改用户
-l new old (改名), -d dir/(改家目录), -g/-G group(改组)
passwd [选项] 用户名      # 设置/修改密码
-l (锁定), -u (解锁), -d (删除密码)
userdel [-r] 用户名      # 删除用户（-r 同时删除家目录和邮件池）

# 用户组
groupadd (-g 2000)(-r) groupname                 #创建组(指定GID)(创建系统组)
groupmod (-n new old)/(-g 2500) groupname        #修改组名/GID
groupdel groupname                               # 删除组（确保组内无用户）
gpasswd -M/a user1,user2 groupname               # 设置组成员列表
gpasswd -d user groupname                        # 从组中移除用户
gpasswd groupname                                # 设置组密码
chown [ -R(递归修改)] 所有者[:组] 文件/目录...       # 修改文件或目录的所有者和组
```

### 权限

```Bash
# 修改文件或目录的权限
chmod [ -R(递归修改)] 权限 文件/目录...
# 权限格式
u/g/o/a -> +/-/= -> r/w/x        # 符号表示
(421)-> 421-> 421-> 421          # 数字表示
4xxx(s)  # SUID 执行时获得所有者权限
2xxx(s)  # SGID 执行时获得所属组权限
1xxx(t)  # Sticky 仅所有者可删
# 常用权限
     u   g   o
755（rwx r-x r-x）                # 脚本文件标准
644（rw- r-- r--）                # 配置文件标准
600（rw- --- ---）                # 私密文件（私钥）
700（rwx --- ---）                # 私有目录（家目录）
1777（rwx rwx rwt）               # /tmp 目录
4755（rws r-x r-x）               # passwd 命令（需要SUID修改 /etc/shadow）
2750（rwx r-s ---）               # 团队共享目录（SGID保证继承组）

# 文件权限第一位：类型
- 普通文件, d 目录, l 链接, b/c 设备, s 套接字, p 管道

# 修改sudo权限配置文件
visudo

# ACL(对特定对象赋予权限)
getfacl file.txt                        # 查看文件 file.txt 的ACL
setfacl -m [u/g/o:xxx:r/w/x] file.txt   # 授予权限
setfacl -R -m u:bob:rx /project/src/    # 递归赋予对目录的权限
setfacl -d -m g:qa:r /shared/reports/   # 设置目录的默认ACL，让新文件自动继承 qa 组的权限
setfacl -x u:alice file.txt             # 移除用户 alice 的ACL条目
```

### 防火墙

```Shell
ufw enable/disable/reload      # 启用（开机自启）/ 禁用 / 重启
ufw status [verbose/numbered]  # 查看状态（默认策略、日志级别/带编号显示规则）
ufw reset                      # 重置（清除所有自定义规则，恢复初始禁用状态）
ufw app list                   # 查看所有已注册应用
ufw app info OpenSSH           # 查看应用详细信息

# 默认策略： 阻止/允许 外部发起连接/主动对外连接
ufw allow ssh -> ufw default allow/deny incoming/outgoing

# 管理访问规则
ufw allow/deny/limit [服务名] / [端口/协议] / [from IP(to any port 端口 proto tcp)]
#   放行/拒绝/限制速率    服务     服务端端口               特定IP访问特定端口

# 场景：Web服务器最小化开放端口
ufw default deny incoming
ufw default allow outgoing
ufw allow ssh                  # 管理端口
ufw allow http                 # 80
ufw allow https                # 443
ufw limit ssh                  # SSH增加限速
ufw enable

# 删除访问规则
ufw delete [allow 22/tcp] / 3

# 日志管理
ufw logging on/off              # 开启(low)/关闭日志
ufw logging low/medium/high     # 日志级别：记录被阻止的包/被阻止+允许的包/所有包
```

### SSH加固

```Shell
ssh连接
ssh (-p 2222)(-J jumpserver) user@hostname  'ls -la /data/'
#    指定端口    使用跳板机                    执行单条命令后退出


密钥登录配置流程

# 1.客户端生成密钥对：~/.ssh/id_rsa（私钥）和id_rsa.pub（公钥）
ssh-keygen -t rsa -b 4096    

# 2.把公钥传到服务器
ssh-copy-id -i ~/.ssh/id_rsa.pub user@服务器IP -p 2222

# 3.SSH安全加固（开多终端修改）
修改配置文件：/etc/ssh/sshd_config
PermitRootLogin no                    # 禁止root直接登录
PasswordAuthentication no             # 禁用密码登录（改用密钥）
PubkeyAuthentication yes              # 开启密钥认证
Port 2222                             # 更改默认22端口（防扫描）
MaxAuthTries 3                        # 最大重试次数
ClientAliveInterval 300               # 空闲超时自动断连
AllowUsers user1 user2                # 白名单用户（仅允许特定用户）
DenyUsers/Groups [user/group]         # 黑名单
  
sshd -t                               # 改完检查语法
systemctl restart sshd                # 改完执行
```

### 日志审计

```Shell
# 登录日志
last                        # 查看最近成功登录记录（含时间和IP）
lastb                       # 查看最近失败的登录尝试（暴力破解看这里）
lastlog                     # 查看所有用户最后一次登录时间

# 系统认证日志
/var/log/auth.log
/var/log/messages
/var/log/secure（认证）
/var/log/maillog

auditd（内核级审计）
systemctl start auditd
配置 /etc/audit/rules.d/audit.rules
auditd -w /etc/passwd -p wa -k identity             # 监控passwd写操作
auditd -w /etc/sudoers -p wa -k sudoers             # 监控sudoers变更
auditd -a always,exit -S unlink -S rmdir -k delete  # 监控删除行为
ausearch -k identity                                # 查看审计结果

AIDE（文件完整性检查）
aide --init
mv /var/lib/aide/aide.db.new.gz /var/lib/aide/aide.db.gz
aide --check            # 定期对比（可配合cron做巡检）
```

> 软件包管理，打包解压文件，备份与同步传输

------

### 打包与解压文件

```Shell
# 将多个文件/目录合并成单个归档文件(.tar)
tar [选项] [归档文件名] [源文件/目录...]
-c/-x/-t             # 打包/解包/查看
-z/-j/-J             # 通过 gzip/bzip/xz 解压
-vf                  # 显示过程，指定归档文件名（必须紧跟参数后）
-C                   # 指定解包到目标目录（如 -C /tmp）

# 单个文件压缩解压(.gz)
gzip [选项] [文件]
gzip -k              # 保留原文件 
gzip -d              # 解压.gz文件

# 目录文件压缩+打包(.zip)
zip [选项] 压缩包名.zip [要压缩的文件/目录...]
zip -r               # 压缩目录必须加 -r
zip -d               # 从压缩包中删除指定文件
zip -l               # 显示压缩包中的文件列表
zip -q               # 不显示过程
# 解压(.zip)
unzip [选项] 压缩包名.zip
unzip -d             # 指定解压到目标目录（如 unzip -d /tmp/）
unzip -o             # 覆盖已存在的文件，不提示
unzip -n             # 不覆盖已存在的文件
```

### 软件包管理

```Shell
# 查看
apt list --installed          # 列出所有已安装的包
apt list --upgradable         # 查看哪些包可以升级
apt search nginx              # 搜索软件包
apt show nginx                # 查看包详细信息
apt-cache depends nginx       # 查看包的依赖关系
dpkg -l                       # 列出所有已安装包
dpkg -L package               # 列出包安装的所有文件
dpkg -S /usr/bin/nginx        # 查某个文件属于哪个包
dpkg -s package               # 查看包详细信息

# 安装升级
apt update                    # 更新软件包索引，必须定期执行
apt install nginx  -y         # 安装软件包
dpkg -i package.deb 
apt install ./xxx.deb         # 安装本地deb包（自动解决依赖）
apt [dist-]upgrade [nginx]    # 升级包（自动处理依赖变化）
apt install -f                # 修复破损依赖
dpkg --configure -a           # 修复未配置完的包
apt-mark hold/unhold nginx    # 锁定/解锁软件升级


# 卸载清理
apt remove nginx -y           # 卸载（保留配置文件）
apt purge nginx -y            # 彻底卸载
apt autoremove -y             # 删除不再需要的依赖包
apt clean                     # 清理下载的包缓存（/var/cache/apt/archives）
apt autoclean                 # 只清理过期版本的包缓存

# 安全补丁专用（重要）
yum --security check-update
yum --security update
```

### 源码编译

```Shell
# 安装
./configure [--prefix=/usr/local/nginx] [--with-http_ssl_module]
make                           # 编译（耗时较长）
make install                   # 安装到目录

# 常用参数
--prefix=/path                # 指定安装目录（便于卸载，直接删目录即可）
--with-xxx                    # 开启某个功能（如SSL）
--without-xxx                 # 关闭某个功能

# 卸载
cd /源码目录 | make uninstall
```

### 备份与同步传输

```Shell
# 增量同步传输（只备份自上次备份以来变化的数据）
rsync [选项] [源目录/文件..] [目标目录/服务器]            
-a            # 归档模式（保留权限、时间戳等）
-v            # 显示过程
-z            # 压缩传输
-P            # 显示进度+支持断点续传
--delete      # 目标目录多出来的文件删掉（镜像同步）
-e ssh        # 远程传输 

# 临时、一次性的小文件传输
scp [目录] [服务器]

# 全盘克隆：备份整个/dev/sda分区到镜像文件
dd if=/dev/sda of=/path/to/backup.img bs=4M status=progress
```

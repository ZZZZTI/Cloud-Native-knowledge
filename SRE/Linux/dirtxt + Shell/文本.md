> 文本内容的编辑与处理

------

### 管道与重定向

```Bash
# 重定向（把命令的“输入/输出”从屏幕改成“文件”）
输入 >  a.txt           # 把结果写进文件（覆盖原有内容）
输入 >> a.txt           # 把结果追加到文件末尾（不覆盖）
命令 <  a.txt           # 把文件内容作为命令的输入
命令 2> a.txt           # 把报错信息写进文件（1是正常结果，2是报错）
命令 > a.txt 2>&1       # 把报错当成正常结果一起处理（常用！）
&& \                   # &&:前命令执行成功才执行后面，\:跨多行书写命令

# 管道（把左边的输出，传给右边当输入）
命令1 | (命令1的输出->)命令2           # grep,awk,sed,wc,sort,uniq,head,tail
命令1 | xargs (命令1的输出->)命令2     # rm,cp,mv,mkdir,chmod,chown,kill,tar
命令1 | tee 文件                     # 屏幕显示结果，同时自动存进文件（查实时日志必备）
```

### 文本内容查看

```shell
# 查看或拼接文件内容
cat [选项] 文件...
cat -n              # 对所有输出行编号（含空行）
cat -b              # 对非空输出行编号
cat -s              # 将连续的空行压缩为一行
cat -T              # 将制表符显示为 ^I
cat -E              # 在每行结尾显示 $
tac [选项] 文件...   # 反向输出文件内容

# 分页查看大文件内容
less [选项] 文件...
less -N             # 显示行号
less -S             # 不换行显示，可左右滚动
less -R             # 保留颜色转义序列（如日志高亮）
less -g             # 高亮显示上次搜索的关键词
# 交互快捷键：空格（下翻一页）、b（上翻一页）、/关键词（搜索）、q（退出）

# 显示文件开头部分（默认前 10 行）
head [选项] 文件...
head -n N           # 显示前 N 行（如 -n 20）
head -c N           # 显示前 N 个字节（如 -c 1024）
head -q             # 多个文件时不显示文件名头

# 显示文件末尾部分（默认后 10 行，运维最常用）
tail [选项] 文件...
tail -n N           # 显示后 N 行（如 -n 100）
tail -f             # 实时跟踪文件新增内容（日志监控神器）
tail -F             # 类似 -f，但文件被轮转后能重新打开（推荐用 -F）
tail -c N           # 显示后 N 个字节
tail --pid=PID      # 当指定进程结束时，自动终止 tail -f
```

### 过滤编辑分析文本

```Shell
# 在文件或管道中 过滤搜索 匹配文本的行
grep [选项] "正则表达式" [文件]
grep -i             # 忽略大小写
grep -v             # 反向匹配（显示不包含的行）
grep -r             # 递归搜索目录下的所有文件
grep -n             # 显示匹配行在文件中的行号
grep -c             # 只统计匹配的行数（不显示具体行）
grep -E             # 使用扩展正则表达式(\转义,$,?)
grep -l             # 只显示包含匹配内容的文件名（不显示行）
grep -o             # 只输出匹配到的部分（而不是整行）
grep -A/B/C N       # 显示匹配行及其后/前/前后 N 行
grep --color=auto   # 高亮显示匹配的字符串


# 流编辑器，对文件进行文本替换、删除、插入等操作
sed [选项] '地址+命令' [文件]
sed -i              # 直接修改源文件（慎用，建议先备份）
sed -i.bak          # 修改前自动备份为 .bak 后缀文件
sed -n              # 取消默认输出（与打印命令配合）
sed -e              # 允许执行多个脚本命令
# 地址
' '                 # 每行
'10'                # 第 10 行
'1,5'               # 第 1 到 5 行
'#abc#'             # 匹配包含 abc 的行(可以用正则)
'$'                 # 最后一行
'1,#abc#'           # 从第 1 行到第一个匹配 abc 的行
# 命令
's#old#new#g,i,2,p' # 全局替换（g:全部替换,i:忽略大小写,2:替换每行第2个匹配项,p:打印）
'd'                 # 删除行
'p'                 # 打印行（需配合 -n）
'a\ 文本'            # 在指定行后追加（append）
'i\ 文本'            # 在指定行前插入（insert）


# 强大的文本分析工具，按列处理（取日志字段、统计等）
awk [选项] '模式 { 动作 }' [文件]
awk -F [符号]        # 指定输入字段分隔符（默认空格分隔）
awk -v              # 定义变量并传入 awk 内部
# 内置变量
'$0'                # 整行内容
'$1-$N'             # 每行的第 1 到第 N 列
'NF'                # 当前行的字段总数（$NF 表示最后一列）
'NR'                # 当前行号
# 模式（过滤）
' '                 # 每行
'/abc/'             # 包含abc的行
'$3 >= 1000'        # 条件成立的行
'NR==1, NR==5'      # 1-5行范围
'BEGIN/END {..}'    # 在处理任何行之前/后执行动作
# 动作（ ; 分隔）
无动作               # 打印整行
{print "数字：",$1}  # 打印，自动换行
{printf "%d\n",$1}  # 打印格式化，\n换行
{sum+=$1;count++}   # 累加数据
{OFS=" , "}         # 修改输出分隔符（默认为空格）
{ if(..) print ..}  # 条件


# 常用正则表达式格式
^                   # 行首
$                   # 行尾
.                   # 任意单个字符
[abc]               # a、b 或 c 中的任意一个
[^abc]              # 除 a、b、c 以外的任意字符
[a-z]               # 小写字母 a 到 z
[A-Z]               # 大写字母 A 到 Z
[0-9]               # 数字 0 到 9
\b                  # 单词边界
*                   # 匹配前面的元素 0 次或多次
+                   # 匹配前面的元素 1 次或多次
?                   # 匹配前面的元素 0 次或 1 次
{n}                 # 匹配前面的元素恰好 n 次
{n,}                # 匹配前面的元素至少 n 次
{n,m}               # 匹配前面的元素至少 n 次，至多 m 次
\                   # 将元字符转义为普通字符，例如 \. 匹配字面点号，\\ 匹配反斜杠
^$                  # 空行
^#                  # 以 # 开头的行（注释）


# 从文件的每一行中提取文本片段
cut [选项] 文件...
cut -d ':,'         # 指定字段分隔符
cut -f 2,4          # 提取第 N 列（如 1,3 或 2-5）
cut -c 1-10         # 提取每行 n-m 个字符（如 1-10）
```

### 修改与转换文本

```Bash
# 统计文件的行数、单词数、字符数
wc [选项] 文件...
wc -l               # 只统计行数（最常用）
wc -w               # 只统计单词数
wc -c               # 只统计字节数
wc -m               # 只统计字符数（与 -c 区别在于多字节字符）
wc -L               # 显示最长行的长度

# 对文本文件内容进行排序
sort [选项] 文件...
sort -n             # 按数值大小排序（而非字典序）
sort -r             # 反向排序（从大到小）
sort -t             # 指定字段分隔符（如 -t:）
sort -k N           # 按第 N 个字段排序（如 -k 3）
sort -u             # 去重（相同行只保留一行）
sort -h             # 对人类可读数字排序（如 1K, 2M）

# 报告或忽略重复行（通常与 sort 配合使用，处理相邻重复）
uniq [选项] [输入文件] [输出文件]
uniq -c             # 在每行前显示该行重复出现的次数
uniq -d             # 只显示重复行（且只显示一次）
uniq -u             # 只显示不重复的行（出现一次的行）
uniq -i             # 忽略大小写比较

# 替换、删除或压缩标准输入中的字符
tr [选项] 字符集1 [字符集2]
tr -d               # 删除所有匹配字符集1的字符
tr -s               # 压缩重复出现的连续字符为单个
tr -c               # 补集，匹配字符集1以外的所有字符
tr -t               # 截断字符集1长度与字符集2相同
```

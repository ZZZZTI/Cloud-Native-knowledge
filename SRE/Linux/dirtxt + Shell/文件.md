> 目录与文件的操作命令
>

------

### 切换与查看目录

```bash
# 显示当前所在目录的绝对路径
pwd

# 切换当前工作目录
cd [目录路径]        
cd ..(/..)          # 切换到上一级（两级）目录
cd ~                # 切换到当前用户的家目录
cd -                # 切换到上一次所在的目录

# 列出目录内容
ls [选项] [目录]
ls -a               # 显示隐藏文件
ls -l(h)            # 显示详细信息(文件大小)
ls -i/-Z            # 显示 inode 号/SELinux 上下文
ls -t               # 按修改时间排序（最新优先）
ls -r               # 文件名反向排序
ls -lt/ltr/lS       # 按新/旧修改时间/文件大小排序
ls -d */            # 只显示目录
ls *.txt            # 列出所有txt文件
ls file?            # 匹配file后跟一个字符
ls [abc]*           # 以a、b或c开头的文件

# 以树形结构显示目录内容
tree [选项] [目录]   
tree -L level       # 限制显示的最大深度
tree -d             # 只显示目录，不显示文件
tree -a             # 显示隐藏文件
tree -h             # 显示文件大小
```

### 创建与删除目录文件

```bash
# 创建目录
mkdir [选项] dir1,dir2
mkdir -p a/b/c       # 递归创建（父目录不存在时自动创建）
mkdir -m 755         # 创建时指定权限
mkdir -v             # 显示创建信息

# 删除空目录
rmdir [选项] dir1 dir2
rmdir -p a/b/c       # 递归删除空目录（父目录也空才删）
rmdir -v             # 显示删除信息

# 创建空文件（已存在则更新修改时间）
touch [选项] a.txt b.txt  
touch -t 20261204    # 设置指定时间戳
touch -r             # 复制时间戳
touch -d "2023-12-4" # 设置日期

# 删除文件或目录（高危操作）
rm [选项] a b c
rm -r                # 递归删除，用于删除目录及其所有内容（必需）
rm -f                # 强制删除，忽略不存在的文件，不提示确认
rm -i                # 交互模式，删除前逐一询问
rm -v                # 显示删除的详细信息
rm *.log             # 删除所有log文件

# 任何人都不能删/改/重命名
chattr +/-i /etc/passwd    # 加/解锁
```

### 复制与移动文件目录

```Bash
# 复制文件或目录
cp [选项] (a.txt b.txt)(a.txt b.txt dir/)
cp -i                # 覆盖前提示确认
cp -r dir1/ dir2/    # 递归复制目录
cp -a                # 保留属性（权限、时间等）
cp -u                # 只在源文件更新时复制
cp -v                # 显示复制过程
cp -p                # 保留文件属性

# 移动或重命名文件
mv [选项] (a.txt b.txt)(a.txt dir/)(dir1/ dir2/)
mv -i                # 覆盖前提示
mv -f                # 强制覆盖，不询问
mv -u                # 只在源文件更新时移动
mv -v                # 显示移动过程
mv -n                # 不覆盖已存在的文件
```

### 查找文件

```Bash
# 在目录树中查找文件或目录
find [路径] [查找条件] -exec [命令] {} \+
find / -name "*.txt"          # 按名称查找（可加通配符）
find / -iname "README"        # -i忽略大小写
find / -type f/d/l            # 按类型查找:普通文件/目录/符号链接
find / -size +100M/-1k/10M    # 按大小查找:大于100M/小于1k/精确10M
find / -mtime -7/+30          # 按修改时间查找:7天内/30天前
find / -atime -1              # 按访问时间查找:1天内
find / -perm 755/644          # 按权限查找:精确权限/至少包含
find / -user root             # 按用户查找
find / -group root            # 按组查找
find . -empty                 # 查找空文件
find . -print                 # 输出绝对路径
find . -maxdepth 1            # 避免递归
find . -name "*.txt" -exec chmod 644 {} \+   # 批量修改所有 .txt 文件权限为 644
```

### 查看文件信息

```Bash
# 识别文件的类型
file [选项] 文件...
file -i              # 输出 MIME 类型（如 text/plain; charset=utf-8）
file -b              # 不显示文件名，只显示类型描述
file -L              # 对符号链接文件，跟踪显示其指向的真实文件类型
file -z              # 尝试查看压缩文件内部的类型

# 查看文件的详细元数据
stat [选项] 文件...
stat -c '%a %s %y %n %U %G'   # 权限八进制 大小 修改时间 文件名 所有者 组

# 估算文件或目录的磁盘使用量
du [选项] [目录...]
du -h                # 人类可读格式（K, M, G）
du -s                # 只显示总计（不列出子项）

# 创建目录符号链接
ln [选项] 目标 链接名
ln -s                # 目录必须用 -s
ln -f                # 强制覆盖已存在的链接文件
ln -n                # 将链接视为普通文件（避免递归链接）
```

### 比较文件

```Bash
# 逐行比较两个文件的不同（常用于配置文件对比）
diff [选项] 文件1 文件2
diff -u              # 以统一格式（unified）输出，最易读，常作补丁
diff -c              # 以上下文格式（context）输出
diff -r              # 递归比较两个目录中的所有文件
diff -q              # 只报告文件是否不同，不显示具体差异
diff -y              # 并排显示两个文件的差异（side-by-side）
diff -w              # 忽略所有空白字符的差异

# 以字节为单位比较两个文件（用于二进制文件）
cmp [选项] 文件1 [文件2]
cmp -l               # 显示所有不同字节的位置和值
cmp -s               # 静默模式，只返回退出状态码（0相同，1不同，2错误）
cmp -i N             # 跳过前 N 个字节再开始比较
```

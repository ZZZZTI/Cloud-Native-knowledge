> 服务管理，定时任务，运行级别，作业与终端，软件包管理

------

### 服务管理

```shell
systemctl

# 查看状态
systemctl status nginx                     # 查看服务状态（是否运行、进程PID、最近日志）
systemctl is-active nginx                  # 只看是否运行（返回 active/inactive，脚本里用）
systemctl is-enabled nginx                 # 只看是否开机自启（返回 enabled/disabled）
systemctl list-units --type=service        # 列出所有正在运行的服务
systemctl list-units --all --type=service  # 列出所有服务（含停止的）

# 起停管理
systemctl start nginx              # 立即启动
systemctl stop/kill nginx          # 优雅/强制停止
systemctl restart nginx            # 重启
systemctl reload nginx             # 平滑重载配置（不中断服务，仅部分服务支持）
systemctl reload-or-restart nginx  # 能reload就reload，不能就restart
systemctl daemon-reload            # 重新加载所有服务配置文件（改了配置文件必执行）

# 开机自启
systemctl enable nginx             # 设置开机自启
systemctl disable nginx            # 取消开机自启
systemctl enable --now nginx       # 设置自启+立即启动（合体技，省一行命令）

# 屏蔽/解屏蔽
systemctl mask nginx               # 彻底锁死服务（任何人都不能启动，软链接到 /dev/null）
systemctl unmask nginx             # 解锁
```

### crond（定时任务）

```Shell
# 定时任务管理
crontab -e                  # 编辑当前用户的定时任务（/var/spool/crontabs/用户名）
crontab -l                  # 查看当前用户的所有定时任务
crontab -r                  # 删除当前用户的所有定时任务
crontab -u 用户名            # 操作指定用户的定时任务
grep CRON /var/log/syslog   # 查看cron任务日志

# 定时任务格式
[*  *  *  *  *]   [命令]/[脚本路径]   [>/>>]  [文件日志]  2>&1
 ┬  ┬  ┬  ┬  ┬
 │  │  │  │  └── 星期（0-7，0和7都代表周日）
 │  │  │  └───── 月份（1-12）
 │  │  └──────── 日期（1-31）
 │  └─────────── 小时（0-23）
 └────────────── 分钟（0-59）

时间
0 1 2             # 具体时间
*                 # 任意值（每分钟/每小时/每天...）
,                 # 枚举多个值，如 1,3,5
-                 # 范围，如 10-12
/                 # 步长，如 */5 表示每5个单位
@reboot           # 系统重启时执行一次
@daily/hourly..   # 每日/时...
```

### at（一次性任务）

```Shell
# 管理 at 命令
at [时间] -f [有任务的文件]    # 创建一次性任务（交互式输入命令，Ctrl+D 结束）
at -l                       # 查看当前用户的所有待执行任务（等价于 at -l）
at -d <编号>                 # 删除指定编号的任务（等价于 at -d <编号>）
at -b                       # 系统负载低于 0.8 时执行任务（等价于 at -b）
at -c <编号>                 # 查看指定任务的详细内容
at -m                       # 任务执行后发送邮件（即使没有输出）
at -v                       # 显示任务将执行的精确时间（安排前）
at -q <队列名>               # 指定任务队列（a-z，字母越靠前优先级越高）
at -t [[CC]YY]MMDDhhmm      # 用固定格式指定时间（如 202607311430
echo "任务" ｜ batch         # 在系统平均负载低于 0.8 时才执行

# at 任务格式
> [/usr/bin/命令]/[脚本]  >>  [文件日志]
> ...
> ...
> <Ctrl+D>   # 按 Ctrl+D 保存并退出

# 时间格式
at 14:30/2:30pm       # 具体时分（24/12小时制）
at 10:00 07/31/26     # 具体日期：2026年7月31日上午10:00
at now/时分 + 2 minutes/hours/days/weeks/months/years # XX时间后的现在/时分
```

### 切换运行级别

```Shell
# 级别
poweroff.target      # 关机
rescue.target        # 单用户救援模式
multi-user.target    # 多用户命令行模式（服务器默认）
graphical.target     # 图形界面模式
reboot.target        # 重启

# 切换
systemctl get-default           # 查看启动模式
systemctl set-default [级别]     # 设置为开机启动（服务器标配）
systemctl isolate [级别]         # 立即切换

# 查看启动耗时
systemd-analyze                  # 看整体启动时间
systemd-analyze blame            # 看每个服务启动耗时（排慢启动的神器！）
systemd-analyze critical-chain   # 看关键链，找出拖慢启动的元凶
```

### 作业与终端

```Shell
tty                   # 查看自己当前在哪个终端
jobs -l               # 看当前终端的后台任务列表（带PID）

# 启动时作业控制
[作业] &               # 丢到后台跑，不免疫终端关闭
nohup [作业] &         # 加 nohup，免疫终端关闭（输出默认存 nohup.out）

# 运行时作业控制
前台任务->Ctrl+Z->bg %1 # 前台作业放后台
fg %1                  # 把编号1的作业拉回前台
disown %1              # 让作业不再属于这个终端，即使终端关闭也不影响
kill [-9] %1           # 杀作业
Ctrl + C               # 终止当前前台进程（SIGINT）
Ctrl + Z               # 暂停当前前台进程（SIGTSTP）
Ctrl + D               # 退出当前终端（EOF）
Ctrl + L               # 清屏（相当于 clear）

# 作业保活
tmux ls                    # 查看所有会话
tmux new -s 会话名          # 新建会话
./backup.sh                # 在里面跑任务
Ctrl+B 再按 D               # 脱离会话（detach）
exit                       # 大胆关终端
tmux attach -t 会话名       # 回到某个会话
tmux kill-session -t 会话名 # 杀掉某个会话
```

### 软件包管理

```Shell
# 查看
apt list --installed          # 列出所有已安装的包
apt list --upgradable         # 查看哪些包可以升级
apt search nginx              # 搜索软件包
apt show nginx                # 查看包详细信息
apt-cache depends nginx       # 查看包的依赖关系
dpkg -l                       # 列出所有已安装包
dpkg -L package               # 列出包安装的所有文件
dpkg -S /usr/bin/nginx        # 查某个文件属于哪个包
dpkg -s package               # 查看包详细信息

# 安装升级
apt update                    # 更新软件包索引，必须定期执行
apt install nginx  -y         # 安装软件包
dpkg -i package.deb 
apt install ./xxx.deb         # 安装本地deb包（自动解决依赖）
apt [dist-]upgrade [nginx]    # 升级包（自动处理依赖变化）
apt install -f                # 修复破损依赖
dpkg --configure -a           # 修复未配置完的包
apt-mark hold/unhold nginx    # 锁定/解锁软件升级


# 卸载清理
apt remove nginx -y           # 卸载（保留配置文件）
apt purge nginx -y            # 彻底卸载
apt autoremove -y             # 删除不再需要的依赖包
apt clean                     # 清理下载的包缓存（/var/cache/apt/archives）
apt autoclean                 # 只清理过期版本的包缓存

# 安全补丁专用（重要）
yum --security check-update
yum --security update
```

### 源码编译

```Shell
# 安装
./configure [--prefix=/usr/local/nginx] [--with-http_ssl_module]
make                           # 编译（耗时较长）
make install                   # 安装到目录

# 常用参数
--prefix=/path                # 指定安装目录（便于卸载，直接删目录即可）
--with-xxx                    # 开启某个功能（如SSL）
--without-xxx                 # 关闭某个功能

# 卸载
cd /源码目录 | make uninstall
```
